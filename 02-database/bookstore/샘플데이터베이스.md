# 북스토어
- 데이터베이스 스키마
```sql
  create table sample_books (
    book_no number(7) 
      constraint books_no_pk primary key,
    book_title varchar2(200) not null,
    book_writer varchar2(100) not null,
    book_genre varchar2(100) not null,
    book_price number(7) 
      constraint books_price_ck check (book_price >= 0),
    book_stock number(6)     default 100
      constraint books_stock_ck check (book_stock >= 0),
    book_status varchar2(50) default '판매중'
      constraint books_status_ck check (book_status in ('판매중', '절판', '입고대기중', '입고지연')),
    book_create_date date    default sysdate
  );


  create table sample_users (
      user_id  varchar2(20)  primary key,
      user_name  varchar2(50)  not null,
      user_password  varchar2(20),
      user_email  varchar2(200),
      user_point  number(10)  default 0,
      user_disabled  char(1)  default 'N',
      user_create_date  date  default sysdate
  );

  create table sample_cart_items (
      user_id  varchar2(20)  references sample_users (user_id),
      book_no  number(6)  references sample_books (book_no),
      cart_create_date  date  default sysdate
  );

  create table sample_orders (
    order_no  number(10)  primary key,
    user_id  varchar2(20)  references sample_users (user_id),
    order_status  varchar2(50)  default '결재완료',          -- '결재대기', '결재완료', '상품준비중', '배송중', '배송완료', '취소'
    order_create_date  date  default sysdate
  );

  create table sample_order_items (
    order_no  number(10)  references sample_orders (order_no),
    book_no  number(6)  references sample_books (book_no),
    order_price  number(7),
    order_amount  number(5)
  );

```
